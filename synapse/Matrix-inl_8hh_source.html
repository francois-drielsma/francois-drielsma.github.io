313 
314  // Cannot compute the determinant of a non square matrix
315  Assert::IsNotEmpty("Matrix<T>::Determinant", "Matrix", _mat);
316  Assert::IsSquare("Matrix<T>::Determinant", "Matrix", _mat);
317 
318  // Get rid of the simple cases without using the heavy artillery
319  if ( Nrows() == 1 ) {
320  return _mat[0][0];

334 template<typename T> Matrix<T> Matrix<T>::EigenVectors(std::vector<T>& lambdas,
335  const double eps) const {
336 
337  // Check that the matrix is square
338  Assert::IsNotEmpty("Matrix<T>::EigenVectors", "Matrix", _mat);
339  Assert::IsSquare("Matrix<T>::EigenVectors", "Matrix", _mat);
340 
341  // Initialize the containers for the decomposition
342  size_t dim = _mat.size(); // Dimension of the square matrix
343  size_t i;
344 
345  std::vector<T> tempvals(dim); // Array to store the temporary eigenvalues
346  for (i = 0; i < dim; i++)
347  tempvals[i] = 1.;
348  std::vector<T> eigenvals(dim);
349 
350  std::vector<std::vector<T>> Mk = _mat; // k^th iteration of the similar matrix
351  Matrix<T> Qk, Q(dim, dim, 0); // k^th iteration of the Q matrix and its composite
352  for (i = 0; i < dim; i++)
353  Q[i][i] = 1;
354 
355  // QR decompose the Matrix M, compute Qk^T*M*Qk, repeat until convergence
356  try {
357  bool done = false;
358  while ( !done ) {
359  DecompQR<T> decQR(Mk);
360  Qk = Matrix<T>(decQR.Q());
361  Q *= Qk;
362  Mk = (Qk.Transpose()*Matrix<T>(Mk)*Qk).std();
363 
364  // Update and check the evolution of the eigenvalues
365  done = true;
366  for (i = 0; i < Mk.size(); i++) {
367  eigenvals[i] = Mk[i][i];
368  if ( fabs((eigenvals[i]-tempvals[i])/tempvals[i]) > eps )
369  done = false;
370  }
371  tempvals = eigenvals;
372  }
373  } catch ( Exceptions::Exception& e ) {
374  throw(Exceptions::Exception(Exceptions::nonRecoverable,
375  "Could not eigendecompose"+std::string(e.what()),
376  "Matrix<T>::EigenVectors"));
377  }
378 
379  // Return the eigen values and vectors
380  lambdas = eigenvals;
381  return Q;
382 }

409 template<typename T> void Matrix<T>::Invert() {
410 
411  // Check that the matrix to invert is square and non-zero dimensional
412  Assert::IsNotEmpty("Matrix<T>::Invert", "Matrix", _mat);
413  Assert::IsSquare("Matrix<T>::Invert", "Matrix", _mat);
414 
415  // Initialize the augmented matrix by adding an identity to it
416  size_t i, j, k;
417  size_t n = Nrows();
418  for (i = 0; i < n; i++) {
419  _mat[i].resize(2*n);
420  _mat[i][n+i] = 1;
421  }
422 
423  // Perform the Gauss-Jordan elimination
424  size_t pivot;
425  T max, temp;
426  for (k = 0; k < n; k++) {
427 
428  // Find the pivot column, the one with the highest k^th element on the
429  pivot = k;
430  max = fabs(_mat[k][k]);
431  for (j = k+1; j < n; j++) {
432  if ( max < fabs(_mat[j][k]) ) {
433  max = fabs(_mat[j][k]);
434  pivot = j;
435  }
436  }
437 
438  // If the the pivot element is 0, return that the matrix is singular, increment otherwise
439  if ( !max ) {
440  throw(Exceptions::Exception(Exceptions::nonRecoverable,
441  "The matrix is singular", "Matrix<T>::Invert"));
442  return;
443  }
444 
445  // Swap the k^th and pivot^th lines if not identical
446  if ( pivot != k )
447  std::swap(_mat[pivot], _mat[k]);
448 
449  // Divide the pivot line by its element on the diagonal
450  temp = _mat[k][k];
451  for (i = 0; i < 2*n; i++)
452  _mat[k][i] /= temp;
453 
454  // Simplify all the lines below the current pivot row
455  for (i = 0; i < n; i++) {
456  if ( i != k ) {
457  temp = _mat[i][k];
458  for (j = k+1; j < 2*n; j++)
459  _mat[i][j] -= temp*_mat[k][j];
460  }
461  _mat[i][k] = 0;
462  }
463  }
464 
465  // Remove the first n columns of the augmented matrix
466  for (i = 0; i < n; i++)
467  _mat[i].erase(_mat[i].begin(), _mat[i].begin()+n);
468 }

538 template<typename T> std::vector<T> Matrix<T>::Solve(const std::vector<T>& b) const {
539 
540  // Check that the matrix to invert is square and non-zero dimensional
541  Assert::IsNotEmpty("Matrix<T>::Solve", "Matrix", _mat);
542  Assert::IsSquare("Matrix<T>::Solve", "Matrix", _mat);
543 
544  // Find the LU decomposition of the matrix. If the matrix is singular, do not bother
545  size_t n = Nrows();
546  DecompLU<T> decLU(_mat);
547  if ( !decLU.Complete() ) {
548  throw(Exceptions::Exception(Exceptions::nonRecoverable,
549  "The matrix is singular", "Matrix<T>::Solve"));
550  return std::vector<T>(n);
551  }
552  std::vector<std::vector<T>> L(decLU.L()), U(decLU.U());
553  std::vector<T> pivots(decLU.Pivots());
554 
555  // Take care of the permutation, define a new b'
556  std::vector<T> bprime(n);
557  size_t i;
558  for (i = 0; i < n; i++)
559  bprime[i] = b[pivots[i]];
560 
561  // Solve Ld = b'
562  std::vector<T> x(n);
563  T s;
564  int j, k;
565  for (k = 0; k < (int)n; k++) {
566  s = 0;
567  for (j = 0; j < k; j++)
568  s += L[k][j]*x[j];
569  x[k] = (bprime[k] - s)/L[k][k];
570  }
571 
572  // Back substitution to solve Ux = d
573  for (k = n-1; k > -1; k--) {
574  s = 0;
575  for (j = k+1; j < (int)n; j++)
576  s += U[k][j]*x[j];
577  x[k] = x[k] - s;
578  }
579 
580  return x;
581 }
